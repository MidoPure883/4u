<script>
  // ======================== قائمة الجمل الرومانسية (الفقاعات) ====================
  const messagesList = [
    "💕 صباح الخير يقمري",
 "😘 أحلى حد يصحي", 
"❤️ بحبك اوي",
 "🌸 حقك عليا ياحببتي",
    "😂 ياجامده",
 "🥺 متزعليش يروحي", 
"🫶 إنتي الحتة بتاعتي",
 "😂 لو مضحكتيش هركبك!",
    "ضحكتك حلوه أوي", 
"😘 بخ", 
"غلطت بس بحبك", 
" يا أجمد بت",
    "❤️ بحب عيونك", 
"💋 بحب شفايفك",
 "😂 بحب طيازك", 
"🙈😂 بحب صدرك",
    "🙈😂 بعشق كسكوسك بقا", 
" بحب كلامك", 
"نفسي أسمع صوتك", 
"💋 موووااااه",
    "إنتي قلبي",
 "مقدرش أزعلك",
 "🥺 وحشتيني", 
"مش هكررها",
    "متجيبي بوسه",
 "مراتي اهي اهي اهي",
 "🥹 سامحيني بقا"
  ];

  let currentScore = 0;
  let spawnInterval = null;
  let gameActive = true;
  const totalMessages = messagesList.length;

  // عناصر DOM
  const gameContainer = document.getElementById('gameContainer');
  const startScreenDiv = document.getElementById('startScreen');
  const scoreDiv = document.getElementById('score');
  const resultDiv = document.getElementById('result');
  const restartBtn = document.getElementById('restartBtn');
  const bgMusic = document.getElementById('bgMusic');
  const popAudio = document.getElementById('popSound');

  // ========== تأثيرات مرئية ==========
  function createBurstEffects(x, y) {
    // انفجار قلوب صغيرة
    for (let i = 0; i < 9; i++) {
      let burstHeart = document.createElement('div');
      burstHeart.className = 'burst';
      burstHeart.innerHTML = ['❤️', '💖', '💗', '💓'][Math.floor(Math.random() * 4)];
      burstHeart.style.left = (x + (Math.random() - 0.5) * 45) + 'px';
      burstHeart.style.top = (y + (Math.random() - 0.5) * 35) + 'px';
      document.body.appendChild(burstHeart);
      setTimeout(() => burstHeart.remove(), 700);
    }
    // نجوم لامعة
    for (let i = 0; i < 7; i++) {
      let star = document.createElement('div');
      star.className = 'star';
      star.innerHTML = '✨';
      star.style.left = (x + (Math.random() - 0.5) * 50) + 'px';
      star.style.top = (y + (Math.random() - 0.5) * 30) + 'px';
      document.body.appendChild(star);
      setTimeout(() => star.remove(), 650);
    }
  }

  // ========== فقاعة الكلام (تظهر عند النقر على القلب) ==========
  function showBubbleMessage(text, clickX, clickY) {
    const bubble = document.createElement('div');
    bubble.className = 'speech-bubble';
    bubble.innerText = text;
    // ضبط مكان الفقاعة قريب من موقع النقر (أعلى قليلاً)
    bubble.style.left = (clickX - 60) + 'px';
    bubble.style.top = (clickY - 65) + 'px';
    document.body.appendChild(bubble);
    
    // إزالة الفقاعة بعد ثانية ونصف
    setTimeout(() => {
      if (bubble && bubble.remove) bubble.remove();
    }, 1600);
  }

  // ========== إنشاء قلب ثلاثي الأبعاد يسقط ==========
  function create3DHeart() {
    if (!gameActive) return;

    const heartDiv = document.createElement('div');
    heartDiv.className = 'heart';
    // إضافة قلب جميل مع تنوع بسيط
    const heartVariants = ['❤️', '💖', '💗', '💘', '💝'];
    heartDiv.innerHTML = heartVariants[Math.floor(Math.random() * heartVariants.length)];
    
    const startX = Math.random() * (window.innerWidth - 80);
    heartDiv.style.left = startX + 'px';
    heartDiv.style.top = '-60px';
    document.body.appendChild(heartDiv);

    let posY = -60;
    let speed = 4.2 + Math.min(currentScore * 0.2, 3.6); // صعوبة تدريجية
    let animationId;
    let isRemoved = false;

    function fall() {
      if (!gameActive || !heartDiv.isConnected) {
        cancelAnimationFrame(animationId);
        return;
      }
      posY += speed;
      heartDiv.style.top = posY + 'px';
      
      if (posY > window.innerHeight + 100) {
        if (heartDiv.isConnected) heartDiv.remove();
        cancelAnimationFrame(animationId);
        return;
      }
      animationId = requestAnimationFrame(fall);
    }
    
    animationId = requestAnimationFrame(fall);
    
    // حدث النقر على القلب (يتم اصطياده)
    heartDiv.onclick = (e) => {
      e.stopPropagation();
      if (!gameActive) return;
      
      // الحصول على موقع النقر الدقيق لعرض الفقاعة والتأثيرات
      const rect = heartDiv.getBoundingClientRect();
      const clickCenterX = rect.left + rect.width / 2;
      const clickCenterY = rect.top + rect.height / 2;
      
      // 1. تشغيل صوت الفقاعة (pop)
      if (popAudio) {
        popAudio.currentTime = 0;
        popAudio.play().catch(err => console.log("pop error", err));
      }
      
      // 2. تأثيرات بصرية قوية
      createBurstEffects(clickCenterX, clickCenterY);
      
      // 3. زيادة النقاط وعرض الفقاعة التي تحتوي على الجملة المناسبة
      if (currentScore < totalMessages) {
        currentScore++;
        scoreDiv.innerHTML = `❤️ ${currentScore} / ${totalMessages}`;
        
        // الحصول على الجملة الترتيبية (أو يمكن جعلها عشوائية، لكن الأفضل الاحتفاظ بالتسلسل العاطفي)
        const messageIndex = currentScore - 1;
        const selectedMessage = messagesList[messageIndex];
        
        // عرض الفقاعة الجميلة عند مكان القلب
        showBubbleMessage(selectedMessage, clickCenterX, clickCenterY);
        
        // فوز؟
        if (currentScore >= totalMessages) {
          gameWin();
        }
      }
      
      // 4. إزالة القلب وإنهاء حركته
      if (heartDiv.isConnected) heartDiv.remove();
      cancelAnimationFrame(animationId);
    };
    
    // تنظيف في حالة إنهاء اللعبة
    heartDiv._cleanup = () => {
      cancelAnimationFrame(animationId);
      if (heartDiv.isConnected) heartDiv.remove();
    };
  }

  // ========== إنهاء اللعبة بالفوز ==========
  function gameWin() {
    if (!gameActive) return;
    gameActive = false;
    
    if (spawnInterval) {
      clearInterval(spawnInterval);
      spawnInterval = null;
    }
    
    // إزالة كل القلوب المتبقية
    document.querySelectorAll('.heart').forEach(heart => {
      if (heart._cleanup) heart._cleanup();
      else heart.remove();
    });
    
    // عرض لوحة النصر
    resultDiv.style.display = 'block';
    resultDiv.innerHTML = '🎉✨ عاااش يميرا! ✨🎉<br>💖 كسبتي كل القلوب 💖';
    restartBtn.style.display = 'block';
  }

  // ========== إعادة ضبط اللعبة بالكامل ==========
  function fullReset() {
    // وقف التوليد
    if (spawnInterval) {
      clearInterval(spawnInterval);
      spawnInterval = null;
    }
    
    // إزالة كل القلوب
    document.querySelectorAll('.heart').forEach(heart => {
      if (heart._cleanup) heart._cleanup();
      else heart.remove();
    });
    
    // إخفاء الفقاعات المتبقية
    document.querySelectorAll('.speech-bubble').forEach(b => b.remove());
    
    // إعادة المتغيرات
    currentScore = 0;
    gameActive = true;
    scoreDiv.innerHTML = `❤️ 0 / ${totalMessages}`;
    resultDiv.style.display = 'none';
    restartBtn.style.display = 'none';
    
    // إعادة spawn القلوب
    spawnInterval = setInterval(() => {
      if (gameActive && currentScore < totalMessages) {
        create3DHeart();
      } else if (currentScore >= totalMessages && gameActive) {
        gameWin();
      }
    }, 1050);
  }

  // ========== بدء اللعبة للمرة الأولى ==========
  function startGameSession() {
    if (spawnInterval) clearInterval(spawnInterval);
    document.querySelectorAll('.heart').forEach(h => h.remove());
    document.querySelectorAll('.speech-bubble').forEach(b => b.remove());
    currentScore = 0;
    gameActive = true;
    scoreDiv.innerHTML = `❤️ 0 / ${totalMessages}`;
    resultDiv.style.display = 'none';
    restartBtn.style.display = 'none';
    
    spawnInterval = setInterval(() => {
      if (gameActive && currentScore < totalMessages) {
        create3DHeart();
      } else if (currentScore >= totalMessages && gameActive) {
        gameWin();
      }
    }, 1020);
  }

  // ========== الانتقال من شاشة البداية إلى اللعبة ==========
  function transitionToGame() {
    // تشغيل الموسيقى بعد تفاعل المستخدم
    bgMusic.play().catch(e => console.log("music autoplay blocked? but user clicked", e));
    
    startScreenDiv.classList.add('hide');
    gameContainer.classList.add('active');
    
    setTimeout(() => {
      startGameSession();
    }, 200);
  }

  // ========== إعادة التشغيل (زر إعادة اللعب) ==========
  function handleRestart() {
    if (!gameActive || currentScore >= totalMessages) {
      fullReset();
    } else {
      if (confirm("✨ هل تريد بدء مغامرة جديدة من القلب؟ ✨")) {
        fullReset();
      }
    }
  }

  // ========== قلوب خلفية متحركة ثلاثية الأبعاد في شاشة البداية ==========
  function spawnBackground3DHearts() {
    setInterval(() => {
      if (startScreenDiv && !startScreenDiv.classList.contains('hide')) {
        const bgHeart = document.createElement('div');
        bgHeart.className = 'floating-heart';
        bgHeart.innerHTML = Math.random() > 0.6 ? '❤️' : '💖';
        bgHeart.style.left = Math.random() * 100 + '%';
        bgHeart.style.fontSize = (22 + Math.random() * 28) + 'px';
        bgHeart.style.animationDuration = 5 + Math.random() * 6 + 's';
        startScreenDiv.appendChild(bgHeart);
        setTimeout(() => bgHeart.remove(), 8000);
      }
    }, 400);
  }

  // ========== ربط الأحداث ==========
  function bindAllEvents() {
    const enterBtn = document.getElementById('enter');
    if (enterBtn) enterBtn.addEventListener('click', transitionToGame);
    restartBtn.addEventListener('click', handleRestart);
    
    // تحسين إعادة ضبط عند تغيير حجم النافذة لا يؤثر على اللعب
    window.addEventListener('resize', () => {});
  }

  // ========== تهيئة الصفحة ==========
  function init() {
    bindAllEvents();
    spawnBackground3DHearts();
    if (bgMusic) bgMusic.volume = 0.45;
    if (popAudio) popAudio.volume = 0.7;
    gameContainer.classList.remove('active');
    startScreenDiv.classList.remove('hide');
  }
  
  init();
</script>
